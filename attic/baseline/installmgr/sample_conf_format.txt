# Sample SWORD .conf file format from ~/.sword/mods.d/

# Bible module example (kjv.conf):
[KJV]
DataPath=./modules/texts/ztext/kjv/
ModDrv=zText
BlockType=BOOK
CompressType=ZIP
SourceType=OSIS
Encoding=UTF-8
Lang=en
Description=King James Version (1769) with Strongs Numbers and Morphology
About=The King James Version of the Holy Bible\par\par This is the 1769 Authorized Version (commonly known as the King James Version, or KJV) with Strong's Numbers added for study and research.
LCSH=Bible. English.
DistributionLicense=Public Domain
TextSource=CCEL
Version=3.1
Feature=StrongsNumbers
Feature=Morphology
GlobalOptionFilter=OSISStrongs
GlobalOptionFilter=OSISMorph
GlobalOptionFilter=OSISLemma
GlobalOptionFilter=OSISRedLetterWords

# Key-Value pairs used:
# - Required: DataPath, ModDrv
# - Common: Lang, Description, Version, SourceType, Encoding
# - Optional: About, LCSH, DistributionLicense, TextSource
# - Features: Feature (can have multiple), GlobalOptionFilter
# - Compression: CompressType, BlockType
